445 template <typename T>
446 template <typename... Args>
447 request_builder_t<T>::request_builder_t(supervisor_t &sup_, actor_base_t &actor_, const address_ptr_t &destination_,
448  const address_ptr_t &reply_to_, Args &&... args)
449  : sup{sup_}, actor{actor_}, request_id{++sup.last_req_id}, destination{destination_}, reply_to{reply_to_},
450  do_install_handler{false} {
451  auto addr = sup.address_mapping.get_addr(actor_, response_message_t::message_type);
452  if (addr) {
453  imaginary_address = addr;
454  } else {
455  // subscribe to imaginary address instead of real one because of
456  // 1. faster dispatching
457  // 2. need to distinguish between "timeout guarded responses" and "responses to own requests"
458  imaginary_address = sup.make_address();
459  do_install_handler = true;
460  }
461  req.reset(new request_message_t{destination, request_id, imaginary_address, std::forward<Args>(args)...});
462 }

464 template <typename T> std::uint32_t request_builder_t<T>::send(pt::time_duration timeout) noexcept {
465  if (do_install_handler) {
466  install_handler();
467  }
468  auto fn = &request_traits_t<T>::make_error_response;
469  sup.request_map.emplace(request_id, request_curry_t{fn, reply_to, req});
470  sup.put(req);
471  sup.start_timer(timeout, request_id);
472  return request_id;
473 }

475 template <typename T> void request_builder_t<T>::install_handler() noexcept {
476  auto handler = lambda<response_message_t>([supervisor = &sup](response_message_t &msg) {
477  auto request_id = msg.payload.request_id();
478  auto it = supervisor->request_map.find(request_id);
479  if (it != supervisor->request_map.end()) {
480  supervisor->cancel_timer(request_id);
481  auto &orig_addr = it->second.reply_to;
482  supervisor->template send<wrapped_res_t>(orig_addr, msg.payload);
483  supervisor->request_map.erase(it);
484  }
485  // if a response to request has arrived and no timer can be found
486  // that means that either timeout timer already triggered
487  // and error-message already delivered or response is not expected.
488  // just silently drop it anyway
489  });
490  auto handler_ptr = sup.subscribe(handler, imaginary_address);
491  sup.address_mapping.set(actor, response_message_t::message_type, handler_ptr, imaginary_address);
492 }
