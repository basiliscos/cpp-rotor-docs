170 template <typename Actor, typename Supervisor, typename... Args>
171 intrusive_ptr_t<Actor> make_actor(Supervisor &sup, Args... args) {
172  using ctor_t = details::actor_ctor_t<Actor, Supervisor>;
173  auto state = sup.get_state();
174  if ((state != state_t::INITIALIZED) && (state != state_t::OPERATIONAL)) {
175  sup.context->on_error(make_error_code(error_code_t::supervisor_wrong_state));
176  }
177  auto actor = ctor_t::construct(&sup, std::forward<Args>(args)...);
178  actor->do_initialize(sup.context);
179  sup.template send<payload::create_actor_t>(sup.get_address(), actor);
180  return actor;
181 }
