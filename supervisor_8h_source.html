388 template <typename LocalDelivery> void delivery_plugin_t<LocalDelivery>::process() noexcept {
389  while (queue->size()) {
390  auto message = queue->front();
391  auto &dest = message->address;
392  queue->pop_front();
393  auto &dest_sup = dest->supervisor;
394  auto internal = &dest_sup == actor;
395  if (internal) { /* subscriptions are handled by me */
396  auto *local_recipients = subscription_map->get_recipients(*message);
397  if (local_recipients) {
398  LocalDelivery::delivery(message, *local_recipients);
399  }
400  } else if (dest_sup.address->same_locality(*address)) {
401  auto *local_recipients = dest_sup.subscription_map.get_recipients(*message);
402  if (local_recipients) {
403  LocalDelivery::delivery(message, *local_recipients);
404  }
405  } else {
406  dest_sup.enqueue(std::move(message));
407  }
408  }
409 }

423 template <typename... Args>
424 request_builder_t<T>::request_builder_t(supervisor_t &sup_, actor_base_t &actor_, const address_ptr_t &destination_,
425  const address_ptr_t &reply_to_, Args &&...args)
426  : sup{sup_}, actor{actor_}, request_id{sup.next_request_id()}, destination{destination_}, reply_to{reply_to_},
427  do_install_handler{false} {
428  auto addr = sup.address_mapping.get_mapped_address(actor_, response_message_t::message_type);
429  if (addr) {
430  imaginary_address = addr;
431  } else {
432  // subscribe to imaginary address instead of real one because of
433  // 1. faster dispatching
434  // 2. need to distinguish between "timeout guarded responses" and "responses to own requests"
435  imaginary_address = sup.make_address();
436  do_install_handler = true;
437  }
438  req.reset(
439  new request_message_t{destination, request_id, imaginary_address, reply_to_, std::forward<Args>(args)...});
440 }

442 template <typename T> request_id_t request_builder_t<T>::send(pt::time_duration timeout) noexcept {
443  if (do_install_handler) {
444  install_handler();
445  }
446  auto fn = &request_traits_t<T>::make_error_response;
447  sup.request_map.emplace(request_id, request_curry_t{fn, reply_to, req});
448  sup.put(req);
449  sup.start_timer(request_id, timeout, sup, &supervisor_t::on_request_trigger);
450  return request_id;
451 }

453 template <typename T> void request_builder_t<T>::install_handler() noexcept {
454  auto handler = lambda<response_message_t>([supervisor = &sup](response_message_t &msg) {
455  auto request_id = msg.payload.request_id();
456  auto it = supervisor->request_map.find(request_id);
457  if (it != supervisor->request_map.end()) {
458  auto &orig_addr = it->second.origin;
459  supervisor->template send<wrapped_res_t>(orig_addr, msg.payload);
460  supervisor->discard_request(request_id);
461  }
462  // if a response to request has arrived and no timer can be found
463  // that means that either timeout timer already triggered
464  // and error-message already delivered or response is not expected.
465  // just silently drop it anyway
466  });
467  auto wrapped_handler = wrap_handler(sup, std::move(handler));
468  auto info = sup.subscribe(wrapped_handler, imaginary_address, &actor, owner_tag_t::SUPERVISOR);
469  sup.address_mapping.set(actor, info);
470 }
