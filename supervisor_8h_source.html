376 template <typename LocalDelivery> void delivery_plugin_t<LocalDelivery>::process() noexcept {
377  while (queue->size()) {
378  auto message = queue->front();
379  auto &dest = message->address;
380  queue->pop_front();
381  auto &dest_sup = dest->supervisor;
382  auto internal = &dest_sup == actor;
383  if (internal) { /* subscriptions are handled by me */
384  auto *local_recipients = subscription_map->get_recipients(*message);
385  if (local_recipients) {
386  LocalDelivery::delivery(message, *local_recipients);
387  }
388  } else if (dest_sup.address->same_locality(*address)) {
389  auto *local_recipients = dest_sup.subscription_map.get_recipients(*message);
390  if (local_recipients) {
391  LocalDelivery::delivery(message, *local_recipients);
392  }
393  } else {
394  dest_sup.enqueue(std::move(message));
395  }
396  }
397 }

410 template <typename T>
411 template <typename... Args>
412 request_builder_t<T>::request_builder_t(supervisor_t &sup_, actor_base_t &actor_, const address_ptr_t &destination_,
413  const address_ptr_t &reply_to_, Args &&... args)
414  : sup{sup_}, actor{actor_}, request_id{sup.next_request_id()}, destination{destination_}, reply_to{reply_to_},
415  do_install_handler{false} {
416  auto addr = sup.address_mapping.get_mapped_address(actor_, response_message_t::message_type);
417  if (addr) {
418  imaginary_address = addr;
419  } else {
420  // subscribe to imaginary address instead of real one because of
421  // 1. faster dispatching
422  // 2. need to distinguish between "timeout guarded responses" and "responses to own requests"
423  imaginary_address = sup.make_address();
424  do_install_handler = true;
425  }
426  req.reset(
427  new request_message_t{destination, request_id, imaginary_address, reply_to_, std::forward<Args>(args)...});
428 }

430 template <typename T> request_id_t request_builder_t<T>::send(pt::time_duration timeout) noexcept {
431  if (do_install_handler) {
432  install_handler();
433  }
434  auto fn = &request_traits_t<T>::make_error_response;
435  sup.request_map.emplace(request_id, request_curry_t{fn, reply_to, req});
436  sup.put(req);
437  sup.start_timer(timeout, request_id);
438  return request_id;
439 }

441 template <typename T> void request_builder_t<T>::install_handler() noexcept {
442  auto handler = lambda<response_message_t>([supervisor = &sup](response_message_t &msg) {
443  auto request_id = msg.payload.request_id();
444  auto it = supervisor->request_map.find(request_id);
445  if (it != supervisor->request_map.end()) {
446  supervisor->cancel_timer(request_id);
447  auto &orig_addr = it->second.origin;
448  supervisor->template send<wrapped_res_t>(orig_addr, msg.payload);
449  supervisor->request_map.erase(it);
450  }
451  // if a response to request has arrived and no timer can be found
452  // that means that either timeout timer already triggered
453  // and error-message already delivered or response is not expected.
454  // just silently drop it anyway
455  });
456  auto wrapped_handler = wrap_handler(sup, std::move(handler));
457  auto info = sup.subscribe(wrapped_handler, imaginary_address, &actor, owner_tag_t::SUPERVISOR);
458  sup.address_mapping.set(actor, info);
459 }
