432 template <typename T>
433 template <typename... Args>
434 request_builder_t<T>::request_builder_t(supervisor_t &sup_, actor_base_t &actor_, const address_ptr_t &destination_,
435  const address_ptr_t &reply_to_, Args &&... args)
436  : sup{sup_}, actor{actor_}, request_id{++sup.last_req_id}, destination{destination_}, reply_to{reply_to_},
437  do_install_handler{false} {
438  auto addr = sup.address_mapping.get_addr(actor_, response_message_t::message_type);
439  if (addr) {
440  imaginary_address = addr;
441  } else {
442  // subscribe to imaginary address instead of real one because of
443  // 1. faster dispatching
444  // 2. need to distinguish between "timeout guarded responses" and "responses to own requests"
445  imaginary_address = sup.make_address();
446  do_install_handler = true;
447  }
448  req.reset(new request_message_t{destination, request_id, imaginary_address, std::forward<Args>(args)...});
449 }

451 template <typename T> std::uint32_t request_builder_t<T>::send(pt::time_duration timeout) noexcept {
452  if (do_install_handler) {
453  install_handler();
454  }
455  auto fn = &request_traits_t<T>::make_error_response;
456  sup.request_map.emplace(request_id, request_curry_t{fn, reply_to, req});
457  sup.put(req);
458  sup.start_timer(timeout, request_id);
459  return request_id;
460 }

462 template <typename T> void request_builder_t<T>::install_handler() noexcept {
463  auto handler = lambda<response_message_t>([supervisor = &sup](response_message_t &msg) {
464  auto request_id = msg.payload.request_id();
465  auto it = supervisor->request_map.find(request_id);
466  if (it != supervisor->request_map.end()) {
467  supervisor->cancel_timer(request_id);
468  auto &orig_addr = it->second.reply_to;
469  supervisor->template send<wrapped_res_t>(orig_addr, msg.payload);
470  supervisor->request_map.erase(it);
471  }
472  // if a response to request has arrived and no timer can be found
473  // that means that either timeout timer already triggered
474  // and error-message already delivered or response is not expected.
475  // just silently drop it anyway
476  });
477  auto handler_ptr = sup.subscribe(handler, imaginary_address);
478  sup.address_mapping.set(actor, response_message_t::message_type, handler_ptr, imaginary_address);
479 }
