340 template <typename Actor, typename Supervisor, typename... Args>
341 intrusive_ptr_t<Actor> make_actor(Supervisor &sup, Args... args) {
342  using ctor_t = details::actor_ctor_t<Actor, Supervisor>;
343  auto state = sup.get_state();
344  auto context = sup.get_context();
345  if ((state != state_t::INITIALIZED) && (state != state_t::OPERATIONAL)) {
346  context->on_error(make_error_code(error_code_t::supervisor_wrong_state));
347  }
348  auto actor = ctor_t::construct(&sup, std::forward<Args>(args)...);
349  actor->do_initialize(context);
350  sup.template send<payload::create_actor_t>(sup.get_address(), actor);
351  return actor;
352 }
